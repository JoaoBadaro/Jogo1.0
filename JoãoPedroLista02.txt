Console.WriteLine("Digite seu sexo (M/F):");
char sexo = Convert.ToChar(Console.ReadLine().ToUpper());


IMC = peso * (altura * altura);



if (sexo == 'M')
{
    if (IMC < 20)
     Console.WriteLine("Abaixo do peso");
    
    else if (IMC < 25)
    Console.WriteLine("Peso ideal");
    
    else
    Console.WriteLine("Acima do peso");
    
}
if (sexo == 'F')
{
    if (IMC < 19)
     Console.WriteLine("Abaixo do peso");

        Console.WriteLine("Peso ideal");

       Console.WriteLine("Acima do peso");
